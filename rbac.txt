USE WAREHOUSE AVIATION_WH;
USE DATABASE AVIATION_DB;
USE SCHEMA RAW;

--create roles
CREATE OR REPLACE ROLE AIRLINE_ADMIN;
CREATE OR REPLACE ROLE ANALYST_ROLE;

--permission

GRANT USAGE ON DATABASE AVIATION_DB TO ROLE ANALYST_ROLE;
GRANT USAGE ON SCHEMA MART TO ROLE ANALYST_ROLE;
GRANT SELECT ON ALL TABLES IN SCHEMA MART TO ROLE ANALYST_ROLE;
GRANT SELECT ON ALL VIEWS IN SCHEMA MART TO ROLE ANALYST_ROLE;

--masking policy

CREATE OR REPLACE MASKING POLICY passport_mask AS
(val STRING) RETURNS STRING ->
CASE
    WHEN CURRENT_ROLE() IN ('AIRLINE_ADMIN')
    THEN val
    ELSE 'XXXXXX'
END;

ALTER TABLE PASSENGER_RAW
MODIFY COLUMN passportNumber
SET MASKING POLICY passport_mask;
