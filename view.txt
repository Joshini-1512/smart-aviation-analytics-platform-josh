USE WAREHOUSE AVIATION_WH;
USE DATABASE AVIATION_DB;
USE SCHEMA CORE;

CREATE OR REPLACE VIEW VW_REVENUE AS
SELECT
    flightId,
    SUM(bookingAmount) AS total_revenue
FROM FACT_BOOKING
GROUP BY flightId;


CREATE OR REPLACE MATERIALIZED VIEW MV_FLIGHT_REVENUE AS
SELECT
    flightId,
    SUM(bookingAmount) AS revenue
FROM FACT_BOOKING
GROUP BY flightId;


SELECT * FROM VW_REVENUE;
SELECT * FROM MV_FLIGHT_REVENUE;
